<script>
  import { onMount } from 'svelte';
  import { fade, fly } from 'svelte/transition';
  import { Chrome } from 'lucide-svelte';

  const tutorialSteps = [
    {
      title: "Install the Extension",
      description: "Add our extension to Chrome with just one click and start saving immediately.",
      image: "/api/placeholder/800/600"
    },
    {
      title: "Browse Normally",
      description: "Continue shopping on your favorite sites. We'll automatically detect when savings are available.",
      image: "/api/placeholder/800/600"
    },
    {
      title: "Save Instantly",
      description: "When we find a better deal, just click to apply the savings to your purchase.",
      image: "/api/placeholder/800/600"
    }
  ];

  const beforeAfterImages = [
    {
      before: "/api/placeholder/400/600",
      after: "/api/placeholder/400/600",
      title: "Sarah saved $150"
    },
    {
      before: "/api/placeholder/400/600",
      after: "/api/placeholder/400/600",
      title: "Mike saved $200"
    },
    {
      before: "/api/placeholder/400/600",
      after: "/api/placeholder/400/600",
      title: "Lisa saved $175"
    }
  ];

  let currentStep = 0;
  let tutorialVisible = false;
  let intervalId;

  function startTutorialCycle() {
    intervalId = setInterval(() => {
      currentStep = (currentStep + 1) % tutorialSteps.length;
    }, 5000);
  }

  onMount(() => {
    const observer = new IntersectionObserver(
      (entries) => {
        entries.forEach(entry => {
          if (entry.isIntersecting) {
            tutorialVisible = true;
            startTutorialCycle();
          } else {
            tutorialVisible = false;
            if (intervalId) clearInterval(intervalId);
          }
        });
      },
      { threshold: 0.3 }
    );

    const tutorialSection = document.querySelector('#tutorial-section');
    if (tutorialSection) {
      observer.observe(tutorialSection);
    }

    return () => {
      if (intervalId) clearInterval(intervalId);
      observer.disconnect();
    };
  });
</script>

<div class="min-h-screen bg-white">
  <!-- Hero Section -->
  <section class="max-w-6xl mx-auto px-4 py-16">
    <div class="text-center">
      <h1 class="text-4xl font-bold text-gray-900 mb-6">
        Save Money Automatically While You Shop
      </h1>
      <p class="text-xl text-gray-600 mb-8">
        Our browser extension finds the best deals so you don't have to
      </p>
      <a
        href="#"
        class="inline-flex items-center px-6 py-3 bg-blue-600 text-white font-semibold rounded-lg hover:bg-blue-700 transition-colors duration-200"
      >
        <Chrome class="mr-2" />
        Add to Chrome - It's Free
      </a>
    </div>
  </section>

  <!-- Tutorial Section -->
  <section id="tutorial-section" class="bg-gray-50 py-16">
    <div class="max-w-6xl mx-auto px-4">
      <h2 class="text-3xl font-bold text-center text-gray-900 mb-12">
        How It Works
      </h2>
      
      {#if tutorialVisible}
        <div class="flex items-center gap-8">
          <div class="w-2/3">
            {#key currentStep}
              <img
                src={tutorialSteps[currentStep].image}
                alt="Tutorial step"
                class="rounded-lg shadow-lg w-full"
                in:fly={{ x: -200, duration: 800 }}
                out:fade
              />
            {/key}
          </div>
          <div class="w-1/3">
            {#key currentStep}
              <div
                in:fly={{ x: 200, duration: 800 }}
                out:fade
              >
                <h3 class="text-2xl font-semibold text-gray-900 mb-4">
                  {tutorialSteps[currentStep].title}
                </h3>
                <p class="text-gray-600">
                  {tutorialSteps[currentStep].description}
                </p>
              </div>
            {/key}
          </div>
        </div>
      {/if}

      <!-- Tutorial Progress Indicators -->
      <div class="flex justify-center mt-8 gap-2">
        {#each tutorialSteps as _, i}
          <button
            class="w-2 h-2 rounded-full transition-colors duration-200 {currentStep === i ? 'bg-blue-600' : 'bg-gray-300'}"
            on:click={() => currentStep = i}
            aria-label="Go to step {i + 1}"
          />
        {/each}
      </div>
    </div>
  </section>

  <!-- Before/After Section -->
  <section class="py-16">
    <div class="max-w-6xl mx-auto px-4">
      <h2 class="text-3xl font-bold text-center text-gray-900 mb-12">
        Real People, Real Savings
      </h2>
      <div class="relative">
        <div class="flex overflow-x-auto space-x-8 pb-8 snap-x">
          {#each beforeAfterImages as item}
            <div class="flex-none w-[800px] snap-center">
              <div class="flex gap-4">
                <div class="relative">
                  <img
                    src={item.before}
                    alt="Before"
                    class="rounded-lg"
                  />
                  <span class="absolute top-4 left-4 bg-white px-4 py-2 rounded-full font-semibold">
                    Before
                  </span>
                </div>
                <div class="relative">
                  <img
                    src={item.after}
                    alt="After"
                    class="rounded-lg"
                  />
                  <span class="absolute top-4 left-4 bg-white px-4 py-2 rounded-full font-semibold">
                    After
                  </span>
                </div>
              </div>
              <p class="text-center text-xl font-semibold mt-4">
                {item.title}
              </p>
            </div>
          {/each}
        </div>
      </div>
    </div>
  </section>
</div>
